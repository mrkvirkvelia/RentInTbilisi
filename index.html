  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <script>

    // Список агентов

    const agents = [

      "Omari Mdzavashvili", "Semyon Petunin", "Artem Petunin", "Maksim Lavilin",

      "Giga Gaganashvili", "Giorgi Beselia", "Zaza Kvitsiani", "Vladislav Ivliev",

      "David Kazakov", "Shorena Jvaridze", "Vitaly Tavrel", "Arkadiy Logunov",

      "Nino Beridze", "Levan Tskitishvili", "Irakli Kapanadze", "Tamuna Kharshiladze",

      "Lasha Gelashvili", "Mariam Tsiklauri", "Giorgi Kalandadze", "Salome Abashidze",

      "Zurab Khutsishvili", "Tamar Dolidze"

    ];


    // Районы с координатами

    const districts = [

      { name: "Ваке", coords: [41.7089, 44.7514] },

      { name: "Сабуртало", coords: [41.7380, 44.7325] },

      { name: "Вера", coords: [41.7061, 44.7794] },

      { name: "Дидубе", coords: [41.7415, 44.7930] },

      { name: "Чугурети", coords: [41.7123, 44.7986] },

      { name: "Авлабари", coords: [41.6931, 44.8158] },

      { name: "Глдани", coords: [41.8048, 44.8357] },

      { name: "Исани", coords: [41.6832, 44.8500] },

      { name: "Варкетили", coords: [41.6983, 44.8794] },

      { name: "Мтацминда", coords: [41.6950, 44.7892] },

      { name: "Крцаниси", coords: [41.6134, 44.9084] }

    ];


    // Получить номер недели в году

    function getWeekNumber() {

      const now = new Date();

      const start = new Date(now.getFullYear(), 0, 1);

      const diff = now - start + ((start.getTimezoneOffset() - now.getTimezoneOffset()) * 60000);

      return Math.floor(diff / (7 * 24 * 60 * 60 * 1000));

    }


    // Перемешивание с фиксированной "солью"

    function seededShuffle(array, seed) {

      let a = array.slice();

      for (let i = a.length - 1; i > 0; i--) {

        seed = (seed * 9301 + 49297) % 233280;

        const j = Math.floor((seed / 233280) * (i + 1));

        [a[i], a[j]] = [a[j], a[i]];

      }

      return a;

    }


    // Распределить агентов по районам (по 2 на район)

    const week = getWeekNumber();

    const shuffledAgents = seededShuffle(agents, week);

    const assignments = districts.map((district, i) => {

      return {

        district: district,

        agents: shuffledAgents.slice(i * 2, i * 2 + 2)

      };

    });


    // Инициализация карты

    const map = L.map('map').setView([41.7151, 44.7931], 12);


    // Подключение базовой карты

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    // Добавление районов и агентов

    const agentListDiv = document.getElementById('agent-list');

    const markers = [];


    assignments.forEach(item => {

      const marker = L.marker(item.district.coords).addTo(map);

      marker.bindPopup(`<b>${item.district.name}</b><br>${item.agents.join('<br>')}`);

      markers.push(marker);


      // Добавление в список

      const districtDiv = document.createElement('div');

      districtDiv.className = 'district';

      const districtName = document.createElement('strong');

      districtName.textContent = item.district.name;

      districtDiv.appendChild(districtName);

      const ul = document.createElement('ul');

      item.agents.forEach(agent => {

        const li = document.createElement('li');

        const agentSpan = document.createElement('span');

        agentSpan.className = 'agent';

        agentSpan.textContent = agent;

        agentSpan.addEventListener('mouseover', () => {

          map.setView(item.district.coords, 14);

          marker.openPopup();

        });

        li.appendChild(agentSpan);

        ul.appendChild(li);

      });

      districtDiv.appendChild(ul);

      agentListDiv.appendChild(districtDiv);

    });

  </script>
