  <script>

    const agents = [

      "Omari Mdzevashvili", "Semyon Petunin", "Artem Petunin", "Maksim Lavilin",

      "Giga Gaganashvili", "Giorgi Beselia", "Zaza Kvitsiani", "Vladislav Ivliev",

      "David Kazakov", "Shorena Jvaridze", "Vitaly Tavrel", "Arkadiy Logunov"

    ];


    const districts = {

      "Vake": [41.7097, 44.7750],

      "Saburtalo": [41.7250, 44.7600],

      "Mtatsminda": [41.6934, 44.8015],

      "Didube": [41.7400, 44.7900],

      "Gldani": [41.7795, 44.8120],

      "Isani": [41.7084, 44.8500]

    };


    // Rotate agents weekly

    function getWeeklyRotation() {

      const week = Math.floor((new Date() - new Date(2024, 0, 1)) / (7 * 24 * 60 * 60 * 1000));

      const rotated = [...agents];

      const offset = (week * 2) % agents.length;

      return rotated.slice(offset).concat(rotated.slice(0, offset));

    }


    const rotatedAgents = getWeeklyRotation();

    const tableBody = document.querySelector("#agentsTable tbody");


    const map = L.map('map').setView([41.7151, 44.8271], 12);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);


    let index = 0;

    for (const district in districts) {

      const coords = districts[district];

      const agent1 = rotatedAgents[index % agents.length];

      const agent2 = rotatedAgents[(index + 1) % agents.length];

      index += 2;


      // Add marker to map

      const marker = L.marker(coords).addTo(map)

        .bindPopup(`<strong>${district}</strong><br>${agent1}<br>${agent2}`);


      // Add row to sidebar

      const row = document.createElement("tr");

      row.innerHTML = `<td>${district}</td><td>${agent1}</td><td>${agent2}</td>`;

      tableBody.appendChild(row);

    }

  </script>
