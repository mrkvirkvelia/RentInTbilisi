<body>

  <div id="map"></div>


  <script src="https://unpkg.com/leaflet/dist/leaflet.js"></script>

  <script>

    // Агентский список

    const agents = [

      "Omari Mdzevashvili", "Semyon Petunin", "Artem Petunin", "Maksim Lavilin",

      "Giga Gaganashvili", "Giorgi Beselia", "Zaza Kvitsiani", "Vladislav Ivliev",

      "David Kazakov", "Shorena Jvaridze", "Vitaly Tavrel", "Arkadiy Logunov"

    ];


    // Районы с координатами

    const districts = [

      { name: "Vake", coords: [41.7095, 44.7758] },

      { name: "Saburtalo", coords: [41.7251, 44.7692] },

      { name: "Didube", coords: [41.7415, 44.7930] },

      { name: "Isani", coords: [41.7158, 44.8400] },

      { name: "Mtatsminda", coords: [41.6935, 44.8015] },

      { name: "Chugureti", coords: [41.7070, 44.8011] }

    ];


    // Получить номер недели в году

    function getWeekNumber() {

      const today = new Date();

      const firstJan = new Date(today.getFullYear(), 0, 1);

      const pastDays = (today - firstJan + 86400000) / 86400000;

      return Math.ceil((firstJan.getDay() + pastDays) / 7);

    }


    // Перемешивание с фиксированной "солонкой"

    function seededShuffle(array, seed) {

      let a = array.slice();

      for (let i = a.length - 1; i > 0; i--) {

        const j = seed % (i + 1);

        [a[i], a[j]] = [a[j], a[i]];

        seed = (seed * 16807) % 2147483647;

      }

      return a;

    }


    const week = getWeekNumber();

    const shuffledAgents = seededShuffle(agents, week);

    const districtAssignments = [];


    for (let i = 0; i < districts.length; i++) {

      const agentPair = [shuffledAgents[i * 2], shuffledAgents[i * 2 + 1]];

      districtAssignments.push({

        ...districts[i],

        agents: agentPair

      });

    }


    const map = L.map('map').setView([41.7151, 44.8271], 12);

    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

      attribution: '&copy; OpenStreetMap contributors'

    }).addTo(map);


    districtAssignments.forEach(d => {

      const agentText = d.agents.join("<br>");

      L.marker(d.coords).addTo(map)

        .bindPopup(`<b>${d.name}</b><br>${agentText}`);

    });

  </script>
